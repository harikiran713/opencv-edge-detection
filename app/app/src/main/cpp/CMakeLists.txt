project("native-lib")

# Use C++17
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find OpenCV package

message(STATUS "OpenCV library status:")

# Link libraries

        native-lib
        ${OpenCV_LIBS}
        ${log-lib}
        ${jnigraphics-lib}
        android
